rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() { return request.auth != null; }
    function isAdmin() { return isSignedIn() && request.auth.token.admin == true; }
    function isTherapist(uid) { return isSignedIn() && request.auth.uid == uid; }

    match /users/{userId} {
      allow read: if isAdmin();
      allow write: if false;
    }

    match /therapists/{therapistId} {
      allow read, update: if isAdmin() || isTherapist(therapistId);
      allow create: if isAdmin();
    }

    match /bookings/{bookingId} {
      allow read: if isAdmin() || (isSignedIn() && request.auth.uid == resource.data.therapistId);
      allow write: if isAdmin();
    }

    match /moods/{moodId} {
      allow read: if isAdmin();
      allow write: if false;
    }

    match /journals/{journalId} {
      allow read, write: if false;
    }
  }
}
